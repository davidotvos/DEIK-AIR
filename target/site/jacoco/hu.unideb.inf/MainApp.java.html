    public static Customer getCustomerObject(String name){
        try( CustomerDAO cDao = new JpaCustomerDAO()){

            for(Customer c : templi){
                if(Objects.equals(c.getName(), name))
                {
                    return c;
                }
            }

            e.printStackTrace();
        }

    }
